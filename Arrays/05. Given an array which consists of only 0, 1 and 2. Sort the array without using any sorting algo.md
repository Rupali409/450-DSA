static int[] sortingWithoutAlgoHelper(int[] nums){

        int low = nums[0];
        int mid = nums[0];
        int high = nums[nums.length-1];

        while(mid < nums.length){

            switch(nums[mid]){
                case 0:
                    if(nums[low] != nums[mid]){
                        int t;
                        t = nums[low];  
                        nums[low] = nums[mid];  
                        nums[mid] = t;
                    }
                    low++;
                    break;
                case 1:

                    break;
                case 2:
                    int t;
                    t = nums[high];  
                    nums[high] = nums[mid];  
                    nums[mid] = t;
                    high--;
                    break;
                default:
                    break;
            }
            mid++;
        }
        return nums;
    }
